create table action(
action_id INT NOT NULL AUTO_INCREMENT,
hero_action_id INT DEFAULT NULL,
run_id INT DEFAULT NULL,
draw_id INT DEFAULT NULL,
health_id INT DEFAULT NULL,
hero_boost_id INT DEFAULT NULL,
power_crystal_id INT DEFAULT NULL,
special_id INT DEFAULT NULL,
stat_id INT DEFAULT NULL,
free_attack_id INT DEFAULT NULL,
stat_type_to_value_id INT DEFAULT NULL,
directed_code TINYINT NOT NULL,
boost_code TINYINT(1) NOT NULL,
action_keep TINYINT(1) NOT NULL,
action_order TINYINT NOT NULL,
upsrt_user INT NOT NULL,
upsrt_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
PRIMARY KEY(action_id),
FOREIGN KEY(hero_action_id) REFERENCES hero_action(hero_action_id),
FOREIGN KEY(run_id) REFERENCES run(run_id),
FOREIGN KEY(draw_id) REFERENCES draw(draw_id),
FOREIGN KEY(health_id) REFERENCES health(health_id),
FOREIGN KEY(hero_boost_id) REFERENCES hero_boost(hero_boost_id),
FOREIGN KEY(power_crystal_id) REFERENCES power_crystal(power_crystal_id),
FOREIGN KEY(special_id) REFERENCES special(special_id),
FOREIGN KEY(stat_id) REFERENCES stat(stat_id),
FOREIGN KEY(free_attack_id) REFERENCES free_attack(free_attack_id),
FOREIGN KEY(stat_type_to_value_id) REFERENCES stat_type_to_value(stat_type_to_value_id),
FOREIGN KEY(directed_code) REFERENCES ref_directed_code(directed_code),
FOREIGN KEY(boost_code) REFERENCES ref_boost_code(boost_code),
FOREIGN KEY(action_keep) REFERENCES ref_keep_code(keep_code),
FOREIGN KEY(upsrt_user) REFERENCES user(user_id)
)
